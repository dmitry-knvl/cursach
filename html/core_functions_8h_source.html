99/*****************************************************************
100Запрашивает у пользователя путь до файла исходных данных
101\return структуру, хранящую в себе путь до файла, его имя и
102расширение
103*****************************************************************/
104FileData getFileData() {
105 bool err = false, wrn = false, needName = true;
106 char s;
107 int i, n = 0;
108 string tname = "", name = "", path = "", tpath = "", folder = "", ext = "";
109 FileData result;
110
111 //Запрашиваем данные до тех пор, пока пользователь не введет корректные
112 do {
113 log(L"Введите путь до обрабатываемого файла",DC);
114 log(L"Путь", INP);
115 cin >> tpath;
116
117 for (i = 0; i < tpath.length(); i++) {
118 s = tpath[i];
119
120 //Проверка на недопустимые символы
121 if (isForbiddenNameSymb(s)) {
122 err = true;
123 wrn = true;
124 log(L"Введённый путь содержит недопустимые символы (> < : \" | ?)", ERR, 0);
125 break;
126 }
127 else err = false;
128
129 //В folder храним название последнего каталога, который может оказаться именем файла
130 if (s == '/' || s == '\\') {
131 folder = "";
132 n++;
133 }
134 else folder += s;
135 }
136
137 //Если последний каталог имеет расширение - признаем его именем файла и соответственно обрабатываем
138 if (getFIleExtention(folder) != "-1") {
139 ext = getFIleExtention(folder);
140
141 for (i = 0; i < folder.length() - ext.length() - 1; i++) {
142 name += folder[i];
143 }
144 needName = false;
145
146 //Если путь состоит только из имени файла то на всякий случай за путь принимаем путь до исполняемого файла программы
147 if (n == 0) {
148 path = getCurrentPath();
149 if (path == "0") {
150 err = true;
151 wrn = true;
152 log(L"Ошибка получения пути текущего каталога", ERR, err ? 1 : 0);
153 }
154 }
155 else {
156 for (i = 0; i < tpath.length() - name.length() - ext.length() - 1; i++) {
157 path += tpath[i];
158 }
159 }
160 }
161 else {
162 for (i = 0; i < tpath.length() - name.length() - ext.length(); i++) {
163 path += tpath[i];
164 }
165 }
166
167 //На всякий случай приводим все слеши в пути к единому виду
168 for (i = 0; i < path.length(); i++) {
169 if (path[i] == '/') path[i] = '\\';
170 }
171
172 //Если пользователь не дописал в конце пути слещ - дописываем
173 if (path[path.length() - 1] != '\\') path += "\\";
174
175 } while (err);
176
177 //Если имя не было получено в составе пути - получаем его, и будем запрашивать, пока не будут введены корректные данные
178 if (needName) {
179 do {
180 log(L"Введите имя обрабатываемого файла", DC, wrn ? 1 : 0);
181 log(L"Имя", INP);
182 cin >> tname;
183
184 wrn = false;
185
186 //Проверка на недопустимые символы
187 for (i = 0; i < tname.length(); i++) {
188 s = tname[i];
189
190 if (isForbiddenNameSymb(s) || s == '\\' || s == '/') {
191 err = true;
192 log(L"Введённое содержит недопустимые символы (> < : \" | ? / \\)", ERR, 0);
193 break;
194 }
195 else err = false;
196 }
197
198 //Если имя содержит расширение - оставляем его как есть, если не содержит - принимаем за расширение .txt
199 if (getFIleExtention(tname) != "-1") {
200 ext = getFIleExtention(tname);
201 for (i = 0; i < tname.length() - ext.length() - 1; i++) {
202 name += tname[i];
203 }
204 }
205 else {
206 name = tname;
207 ext = "txt";
208 wrn = true;
209 log(L"Введённое имя файла не содержит расширение. По умолчанию было установлено расширение txt", WARN, err?1:0);
210 }
211 } while (err);
212 }
213
214 //Проверяем файл на существование, и в случае, если его нет - рекурсивно запрашиваем новые данные
215 if (!isFileExist(path + name + "." + ext)) {
216 log(L"Файл не найден по пути \"" + to_wstring(path) + to_wstring(name) + L"." + to_wstring(ext) + L"\"", ERR, err||wrn ? 1 : 0);
217 result = getFileData();
218 }
219 else {
220 log(L"Файл \"" + to_wstring(path) + to_wstring(name) + L"." + to_wstring(ext) + L"\" обнаружен",LOG, wrn ? 1 : 0);
221 result.name = name;
222 result.path = path;
223 result.extension = ext;
224 }
225
226 return result;
227}

323/*****************************************************************
324Сортирует массив (вектор) методом Шелла
325\param A указатель на сортируемый массив
326*****************************************************************/
327void shell(vector<wstring>& A) {
328 int i, j, n = A.size(), d = n / 2;
329 wstring t;
330
331 while (d > 0) {
332 for (i = 0; i < n - d; i++) {
333 j = i;
334 while (j >= 0 && A[j] > A[j + d]) {
335 t = A[j];
336 A[j] = A[j + d];
337 A[j + d] = t;
338 j--;
339 }
340 }
341 d = d / 2;
342 }
343}
